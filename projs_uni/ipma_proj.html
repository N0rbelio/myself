matriz = []    
# Leitura da matriz
for i in range(5):
    linha = []
    for j in range(5):
        elemento = float(input(f"Insira o elemento M[{i+1}][{j+1}]: "))
        linha.append(elemento)
    matriz.append(linha)

# Cálculo do maior elemento de cada linha
maiores_elementos_linha = [max(linha) for linha in matriz]

# Cálculo da média dos elementos de cada coluna
medias_coluna = [sum(coluna) / 5 for coluna in zip(*matriz)]

# Exibição dos resultados
print("\nMaior elemento de cada linha:")
for i, maior_elemento in enumerate(maiores_elementos_linha, start=1):
    print(f"Na linha {i}: {maior_elemento}")

print("\nMédia dos elementos de cada coluna:")
for j, media_coluna in enumerate(medias_coluna, start=1):
    print(f"Na coluna {j}: {media_coluna:.2f}")
